#include <stdio.h>
#include <math.h>

float calcularVolumeEsfera(float raio) {
    float volume = (4.0 / 3.0) * M_PI * pow(raio, 3);
    return volume;
}

int main() {
    float raio, volume;

    printf("Digite o raio da esfera: ");
    scanf("%f", &raio);

    volume = calcularVolumeEsfera(raio);

    printf("O volume da esfera Ã©: %.2f\n", volume);


}
